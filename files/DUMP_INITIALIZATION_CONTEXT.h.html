    long (*WriteRoutine)(LARGE_INTEGER *, MDL *);
    void (*FinishRoutine)(void);

    void *MappedRegisterBase;
    void *PortConfiguration;
    unsigned char CrashDump;
    unsigned long MaximumTransferSize;
    unsigned long CommonBufferSize;
    void *TargetAddress;
    long (*WritePendingRoutine)(long, LARGE_INTEGER *, MDL *, void *);

    union
    {
        struct
        {
            unsigned long Signature;
            unsigned long CheckSum;
        } Mbr;
        struct

        } Gpt;
    } DiskInfo;

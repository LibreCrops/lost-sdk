typedef struct _CONFIGURATION_COMPONENT
{
    CONFIGURATION_CLASS Class;
    CONFIGURATION_TYPE Type;
    DEVICE_FLAGS Flags;
    USHORT Version;
    USHORT Revision;
    ULONG Key;
    union
    {
        ULONG AffinityMask;
        struct
        {
            USHORT Group;
            USHORT GroupIndex;
        };
    };
    ULONG ConfigurationDataLength;
    ULONG IdentifierLength;
    CHAR *Identifier;
} CONFIGURATION_COMPONENT, *PCONFIGURATION_COMPONENT;

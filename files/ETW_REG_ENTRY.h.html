typedef struct _ETW_REG_ENTRY
{
    LIST_ENTRY RegList;
    ETW_GUID_ENTRY *GuidEntry;
    USHORT Index;
    USHORT Flags;
    UCHAR EnableMask;
    union
    {
        ULONG SessionId;
        ETW_REPLY_QUEUE *ReplyQueue;
        ETW_REG_ENTRY *ReplySlot[4];
    };
    union
    {
        EPROCESS *Process;
        VOID *Callback;
    };
    VOID *CallbackContext;
} ETW_REG_ENTRY, *PETW_REG_ENTRY;

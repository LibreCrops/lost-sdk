typedef struct _BLOB
{
    union
    {
        LIST_ENTRY ResourceList;
        SLIST_ENTRY FreeListEntry;
    };
    union
    {
        struct
        {
            UCHAR ReferenceCache : 1;
            UCHAR Lookaside : 1;
            UCHAR Initializing : 1;
            UCHAR Deleted : 1;
        } s1;
        UCHAR Flags;
    } u1;
    UCHAR ResourceId;
    SHORT CachedReferences;
    volatile LONG ReferenceCount;
    EX_PUSH_LOCK Lock;
} BLOB, *PBLOB;

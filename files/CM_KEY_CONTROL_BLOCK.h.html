        CM_KEY_HASH KeyHash;

            ULONG ConvKey;

            ULONG KeyCell;
        };

    };

    CM_INTENT_LOCK KeyLock;
    CHILD_LIST TransValueCache;
    CM_TRANS *TransValueListOwner;

} CM_KEY_CONTROL_BLOCK, *PCM_KEY_CONTROL_BLOCK;

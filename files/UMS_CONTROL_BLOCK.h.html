{
    RTL_UMS_CONTEXT *UmsContext;
    SINGLE_LIST_ENTRY *CompletionListEntry;
    KEVENT *CompletionListEvent;
    ULONG ServiceSequenceNumber;
    union
    {
        struct
        {
            KQUEUE UmsQueue;
            LIST_ENTRY QueueEntry;
            RTL_UMS_CONTEXT *YieldingUmsContext;
            VOID *YieldingParam;
            VOID *UmsTeb;
            union
            {
                ULONG PrimaryFlags;
                ULONG UmsContextHeaderReady : 1;
            };
        };
        struct
        {
            KQUEUE *UmsAssociatedQueue;
            LIST_ENTRY *UmsQueueListEntry;
            KUMS_CONTEXT_HEADER *UmsContextHeader;
            KGATE UmsWaitGate;
            VOID *StagingArea;

            {
                volatile LONG Flags;

                {
                    volatile ULONG UmsForceQueueTermination : 1;
                    volatile ULONG UmsAssociatedQueueUsed : 1;
                    volatile ULONG UmsThreadParked : 1;
                    volatile ULONG UmsPrimaryDeliveredContext : 1;
                    volatile ULONG UmsPerformingSingleStep : 1;
                };
            };
        };
    };

} UMS_CONTROL_BLOCK, *PUMS_CONTROL_BLOCK;

{
    USHORT Signature;
    USHORT Flags;
    LARGE_INTEGER LastWriteTime;
    ULONG Spare;
    ULONG Parent;
    ULONG SubKeyCounts[2];
    union
    {
        struct
        {
            ULONG SubKeyLists[2];
            CHILD_LIST ValueList;
        };
        CM_KEY_REFERENCE ChildHiveReference;
    };
    ULONG Security;
    ULONG Class;
    ULONG MaxNameLen : 16;
    ULONG UserFlags : 4;
    ULONG VirtControlFlags : 4;
    ULONG Debug : 8;
    ULONG MaxClassLen;
    ULONG MaxValueNameLen;
    ULONG MaxValueDataLen;
    ULONG WorkVar;
    USHORT NameLength;
    USHORT ClassLength;
    WCHAR Name[1];
} CM_KEY_NODE, *PCM_KEY_NODE;

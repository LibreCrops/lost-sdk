typedef struct _KINTERRUPT
{
    SHORT Type;
    SHORT Size;
    LIST_ENTRY InterruptListEntry;
    UCHAR (*ServiceRoutine)(KINTERRUPT *, VOID *);
    UCHAR (*MessageServiceRoutine)(KINTERRUPT *, VOID *, ULONG);
    ULONG MessageIndex;
    VOID *ServiceContext;
    ULONGLONG SpinLock;
    ULONG TickCount;
    ULONGLONG *ActualLock;
    VOID (*DispatchAddress)(VOID);
    ULONG Vector;
    UCHAR Irql;
    UCHAR SynchronizeIrql;
    UCHAR FloatingSave;
    UCHAR Connected;
    ULONG Number;
    UCHAR ShareVector;
    CHAR Pad[3];
    KINTERRUPT_MODE Mode;
    KINTERRUPT_POLARITY Polarity;
    ULONG ServiceCount;
    ULONG DispatchCount;
    ULONGLONG Rsvd1;
    KTRAP_FRAME *TrapFrame;
    VOID *Reserved;
    ULONG DispatchCode[4];
} KINTERRUPT, *PKINTERRUPT;
